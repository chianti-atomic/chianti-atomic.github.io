temperature_array_long = np.logspace(5,8,10)
density = 1e+9
wavelength = np.linspace(200,400,1000)
wavelength_range = [wavelength[0], wavelength[-1]]
min_abund = 1.e-4
ion_list = ['fe_15', 'fe_16']
